<h3 id="heading-14">代码块</h3>
<pre><code class="language-javascript" data-language="javascript">// React Hook 示例
import React, { useState, useEffect } from 'react';

function BlogPost({ postId }) {
  const [post, setPost] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    async function fetchPost() {
      try {
        const response = await fetch(`/api/posts/${postId}`);
        const postData = await response.json();
        setPost(postData);
      } catch (error) {
        console.error('Failed to fetch post:', error);
      } finally {
        setLoading(false);
      }
    }

    fetchPost();
  }, [postId]);

  if (loading) {
    return &#x3C;div>加载中...&#x3C;/div>;
  }

  return (
    &#x3C;article>
      &#x3C;h1>{post?.title}&#x3C;/h1>
      &#x3C;div dangerouslySetInnerHTML={{ __html: post?.content }} />
    &#x3C;/article>
  );
}
</code></pre>

<h3 id="heading-14">代码块</h3>
<pre><code class="language-javascript" data-language="javascript">// React Hook 示例
import React, { useState, useEffect } from 'react';

function BlogPost({ postId }) {
  const [post, setPost] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    async function fetchPost() {
      try {
        const response = await fetch(`/api/posts/${postId}`);
        const postData = await response.json();
        setPost(postData);
      } catch (error) {
        console.error('Failed to fetch post:', error);
      } finally {
        setLoading(false);
      }
    }

    fetchPost();
  }, [postId]);

  if (loading) {
    return &#x3C;div>加载中...&#x3C;/div>;
  }

  return (
    &#x3C;article>
      &#x3C;h1>{post?.title}&#x3C;/h1>
      &#x3C;div dangerouslySetInnerHTML={{ __html: post?.content }} />
    &#x3C;/article>
  );
}
</code></pre>
